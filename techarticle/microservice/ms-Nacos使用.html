# Nacos+SpringCloud

# 第一步 下载 Nacos

Nacos Server下载地址：

https://github.com/alibaba/nacos/releases

# 上传到linux，并解压缩

1、上传到 /work，然后解压缩

cd  /work

unzip  nacos-server-1.1.3.zip

2、或者，从虚拟机的共享目录，复制到 /work目录，然后解压缩：

cp  /vagrant/nacos-server-1.1.3.zip  /work/
cd  /work
unzip  nacos-server-1.1.3.zip

如果unzip命令不存在，则安装unzip命令

yum install -y unzip zip

# 启动服务器

  • 单机模式运行启动方法：

    /work/nacos/bin/startup.sh -m standalone

  • 修改配置文件 nacos\bin\startup.cmd

    set MODE="standalone"

如果在windows服务器，也就启动的命令不同而已

startup.cmd -m standalone

# 设置Nacos开机启动

编辑启动配置文件 /etc/rc.local，加入开机启动项，里面添加内容：

# 开机启动 Nacos
/usr/bin/su  - root  -c   "/work/nacos/bin/startup.sh -m standalone"

# 通过@Value注解获取配置文件值

  • BoostStrap.properties优先级高于application.yaml
  • nacos字段会覆盖bootstrap.yaml
@RestController
@RequestMapping("/nacosConfig")
@RefreshScope
public class NacosConfigController {

    @Value("${nacos.username}")
    String username;

    @Value("${nacos.password}")
    String password;

    @GetMapping("/getConfigInfo")
    public String info() {
        return "username:"+username + "\n" + "password:"+password;
    }
}

# 添加服务类

package org.dv.nacosorder.feign;


import org.springframework.cloud.openfeign.FeignClient;
import org.springframework.stereotype.Component;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;

@Component
@FeignClient("nacos-stock")//对应服务名称
public interface StockFeignService {

    @RequestMapping("/stock/save")
    String save(@RequestParam("info") String info);
}

# 通过接口调用服务类

package org.dv.nacosorder.controller;


import org.dv.nacosorder.feign.StockFeignService;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import javax.annotation.Resource;

@RestController
@RequestMapping("/order")
public class OrderController {

    @Resource
    StockFeignService stockFeignService;

    @RequestMapping("/test")
    public String test(@RequestParam String info) {
        return stockFeignService.save(info);
    }

}

# 启动类添加注解EnableFeignClients

package org.dv.nacosorder;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.cloud.openfeign.EnableFeignClients;

@SpringBootApplication
@EnableFeignClients(basePackages = "org.dv.nacosorder.feign")
public class NacosOrderApplication {

    public static void main(String[] args) {
        SpringApplication.run(NacosOrderApplication.class, args);
    }

}
