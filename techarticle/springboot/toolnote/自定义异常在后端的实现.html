# 自定义异常在后端的实现

# 1.自定义异常类型

# 继承RuntimeException类，重写构造方法，其实主要需要编写的就是构造方法。

@ControllerAdvice
public class userServiceException extends RuntimeException{
    public userServiceException() {
        super();
    }

    public userServiceException(String message) {
        super(message);
    }
}

# 2.自定义ExceptionHandler类，用于处理方法

package manageserve.exception.handler;

import manageserve.common.Code;
import manageserve.common.Result;
import manageserve.exception.userServiceException;
import org.springframework.web.bind.annotation.ControllerAdvice;
import org.springframework.web.bind.annotation.ExceptionHandler;

@ControllerAdvice
public class ServiceExceptionHandler {

    @ExceptionHandler(userServiceException.class)
    public Result handleServiceException(userServiceException serviceException){
        return Result.error(Code.CODE_500,serviceException.getMessage());
    }
}
