# 3.自定义mapper接口

  • # 新建src/main/java/mapper包，编写查询类实现业务逻辑。

    @Mapper
public interface ItemMapper extends BaseMapper<Item> {
    //插入一天item记录
    public int InsertItem(Item item);
    //插入useritem
    int InsertUserItem(String user_id, String item_id);
    //根据name，foundtime，tpye 返回item列表(不包括本人发布的失物招领)
    List<Item> getItemList(String name, String found_time, String type, String user_id);
    //item是否存在
    boolean isItemExist(String itemId);

    boolean isItemRedempted(String itemId);

    int redemption(Redemption redemption);

    boolean getItemStatus(String itemId);

    int claimOne(String itemId);
}

  • # src/main/resources/mapper新建mapper包。

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.manageserve.mapper.IUserMapper">
    <update id="Update">

        update sys_user
        <set>
            <if test="username != null">
                username=#{username},
            </if>
            <if test="password != null">
                password=#{password},
            </if>
            <if test="nickname != null">
                nickname=#{nickname},
            </if>
            <if test="email != null">
                email=#{email},
            </if>
            <if test="phone != null">
                phone=#{phone},
            </if>
            <if test="address != null">
                address=#{address}
            </if>
        </set>
        <where>
            id=#{id}
        </where>
    </update>


    <select id="selectPage">
        select * from sys_user
        where username like #{username} and email like #{email} and address like #{address}
        order by id desc
        limit #{pageNum},#{pageSize}
    </select>

    <select id="selectTotal">
        select count(*)
        from sys_user
        where username like #{username} and email like #{email} and address like #{address}
    </select>
    <select id="getPasswordByName">
        select password
        from sys_user
        where username=#{username}
    </select>
    <select id="isUserExist">
        select count(*)
        from sys_user
        where username=#{username}
    </select>
</mapper>
