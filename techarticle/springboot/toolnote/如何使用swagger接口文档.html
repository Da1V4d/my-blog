# 如何使用swagger接口文档

# 1.引入maven依赖“springdoc”

<dependency>
	<groupId>org.springdoc</groupId>
	<artifactId>springdoc-openapi-starter-webmvc-ui</artifactId>
	<version>2.2.0</version>
</dependency>

# 2.打开swagger网页调试接口

http://localhost:9090/swagger-ui/index.html#/

# 3.配置接口swagger注释

@Tag(name = "用户管理api",description = "userController")
@RestController
@RequestMapping("/user")
public class userContorller {

    @Autowired()
    IUserService iUserService;


    /*查询所有用户信息*/
    @Operation(summary = "查询所有用户",description = "不需要输入，查询返回所有用户信息")
    @GetMapping("/")
    Result getAll(){
        return Result.success(iUserService.getAll());
    }
    
    
    
    @Operation(summary = "分页查询",description = "输入pageNUm，pageSize，username，email，address")
    @Parameters({
            @Parameter(name = "pageNum",description = "页面编号",required = true),
            @Parameter(name = "pageSize",description = "每页条数",required = true),
            @Parameter(name = "username",description = "用户名",allowEmptyValue = true),
            @Parameter(name = "email",description = "邮箱",allowEmptyValue = true),
            @Parameter(name = "address",description = "地址",allowEmptyValue = true)

    })//allowEmpty可以设置传入参数为空
    @GetMapping("/page")
    Result findPage(
            @RequestParam Integer pageNum,
            @RequestParam Integer pageSize,
            @RequestParam String username,
            @RequestParam String email,
            @RequestParam String address){

        username="%"+username+"%";
        address="%"+address+"%";
        email="%"+email+"%";
        pageNum=(pageNum-1)*pageSize;
        //System.out.println("GetMapping Page");
        return Result.success(iUserService.selectPage(pageNum,pageSize,username,email,address));
    }
}
