# 2、实现分页查询接口

@Operation(summary = "分页查询",description = "输入pageNUm，pageSize，username，email，address")
@Parameters({
        @Parameter(name = "pageNum",description = "页面编号",required = true),
        @Parameter(name = "pageSize",description = "每页条数",required = true),
        @Parameter(name = "username",description = "用户名"),
        @Parameter(name = "email",description = "邮箱"),
        @Parameter(name = "address",description = "地址")

})
@GetMapping("/page")
Map<String,Object> findPage(
        @RequestParam Integer pageNum,
        @RequestParam Integer pageSize,
        @RequestParam String username,
        @RequestParam String email,
        @RequestParam String address){

    username="%"+username+"%";
    address="%"+address+"%";
    email="%"+email+"%";
    pageNum=(pageNum-1)*pageSize;
    //System.out.println("GetMapping Page");
    return iUserService.selectPage(pageNum,pageSize,username,email,address);
}
