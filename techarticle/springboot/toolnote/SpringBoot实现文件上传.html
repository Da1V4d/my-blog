# Spring实现文件上传

# 1、配置yml文件

spring:
  servlet:
    multipart:
      max-file-size: 10MB
      max-request-size: 20MB

# 2、配置接口控制类

public class ItemController {
    @Autowired
    ItemService itemService;

    @PostMapping("/release")
    public Result releaseItem(Item item, @RequestParam("file") MultipartFile file,@RequestHeader String token) throws IOException {
        if(file.isEmpty()){
            return Result.error(Code.CODE_400,"文件为空");
        }
        return itemService.releaseItem(item,token,file);

    }
}

# 3、配置服务类

@Service
public class ItemService implements iItemService {
    @Autowired
    ItemMapper itemMapper;
    @Autowired
    RedemptionMapper redemptionMapper;

    private static final String PHOTO_FILE="C:\\IdealProject\\JavaEEStudy\\LNF\\src\\main\\resources\\static\\item_photo\\";
    //发布失物招领接口item，token，file（物品图片）
    @Override
    public Result releaseItem(Item item,String token, MultipartFile file){
        String filename=file.getOriginalFilename();
        String newName= UUID.randomUUID().toString()+"."+filename.substring(filename.lastIndexOf(".")+1);
        File dest_file=new File(PHOTO_FILE+newName);//目标文件
        try {
            file.transferTo(dest_file);
            item.setPhoto(dest_file.getPath());
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
        item.init();
        item.setPhoto(dest_file.getAbsolutePath());
        Claim Idclaim = JwtUtils.getClaims(token).get("id");
        String id=Idclaim.asString();
        try {
            int count=0;
            count+=itemMapper.InsertItem(item);
            count+=itemMapper.InsertUserItem(id,item.getId());
            String msg=count+" rows affected";
            return Result.success("success "+msg);
        } catch (Exception e) {
            e.printStackTrace();
            return Result.error(Code.CODE_500,"internal serve error");
        }
    }
}
