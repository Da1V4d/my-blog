# SpringBoot进阶

# 一、配置高级

  • 临时属性设置-解决端口冲突问题

  • 通过临时属性修改端口

     java -jar SpringBootLearning-0.0.1-SNAPSHOT.jar  --server.port=8081

  • 修改日志属性

     java -jar SpringBootLearning-0.0.1-SNAPSHOT.jar  --logging.level.org.springframework.web=debug

# 8、完整日志输出配置文件

logdir: src/log/

logging:
  pattern:
    console: "%d ---[%16t] %c %p %m %n"
    file: "%d  ---[%16t] %c %p  %m %n"
  group:
    controller: org.dv.springbootlearning.controller
  level:
    root: info
    controller: info
  file:
    path: src/log/
    name: ${logdir}springboot-learning.log
  logback:
    rollingpolicy:
      max-file-size: 1KB
      file-name-pattern: ${logdir}springboot-learning-%d{yyyy-mm-dd}-%i.log
      #文件名前缀和name相同方便拆分
